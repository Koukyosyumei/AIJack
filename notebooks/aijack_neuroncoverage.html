def fix_seed(seed):
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)
    torch.backends.cudnn.deterministic = True


fix_seed(42)

class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()
        self.fla = nn.Flatten()
        self.fc = nn.Linear(112 * 92, 40)

    def forward(self, x):
        x = self.fla(x)
        x = self.fc(x)
        x = F.softmax(x, dim=1)
        return x


def split_dataloader(data_loader, k):
    dataset = data_loader.dataset
    dataset_size = len(dataset)
    batch_size = data_loader.batch_size

    # Calculate the size of each subset
    subset_size = dataset_size // k
    remainder = dataset_size % k

    # Create a list to store the k DataLoaders
    dataloaders = []

    # Create subsets and DataLoaders
    start_idx = 0
    for i in range(k):
        end_idx = start_idx + subset_size + (1 if i < remainder else 0)
        indices = list(range(start_idx, end_idx))
        sampler = SubsetRandomSampler(indices)
        dataloader = DataLoader(dataset, batch_size=batch_size, sampler=sampler)
        dataloaders.append(dataloader)
        start_idx = end_idx

    return dataloaders

    running_loss = 0
    data_size = 0
    for i, data in enumerate(trainloader, 0):
        # get the inputs; data is a list of [inputs, labels]
        inputs, labels = data

        # zero the parameter gradients
        optimizer.zero_grad()

        # forward + backward + optimize
        outputs = net(inputs)
        loss = criterion(outputs, labels.to(torch.int64))
        loss.backward()
        optimizer.step()

        running_loss += loss.item()
        data_size += inputs.shape[0]

    print(f"epoch {epoch}: loss is {running_loss/data_size}")
